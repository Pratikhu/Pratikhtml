Prime Number Check in C

WAP for finding if a number is prime or not


#include <stdio.h>
#include <conio.h>

void main() {
    int num, i, count = 0;
    clrscr();
    printf("Input a number: ");
    scanf("%d", &num);

    for (i = 1; i <= num; i++) {
        if (num % i == 0)
            count++;
    }

    if (count == 2)
        printf("Prime");
    else
        printf("Not Prime");

    getch();
}

WAP for finding if input number is palindrome or not


#include <stdio.h>

int main() {
    int num,a,rev=0,count;
    printf("Input a number:");
    scanf("%d", &num);
    count=num;
    while(num!=0)
    {
        a=num%10;
        rev=rev*10+a;
        num=num/10;
    }
    if(rev==count)
{
    printf("%d is a palindrome",count);
    else printf("%d is not a palindrome",count);

}
  
return 0;
}
